country_event = {

	title = germany.1.t
	desc = germany.1.d

	option = {

		}

			target = FRA

		}

			add_to_war = { targeted_alliance = FRA enemy = GER } 
		}
		LUX = {
			add_to_war = { targeted_alliance = FRA enemy = GER } 
		}
	}
	
	option = {

		ai_chance = { factor = 20 }

			country_exists = FRA

				has_war_with = HOL
			}
			GER = { has_completed_focus = GER_the_schlieffen_plan }
		}
		create_wargoal = {
			target = FRA
			type = topple_government
		}
		declare_war_on = {
			target = FRA
			type = topple_government
		}
		BEL = {
			add_to_war = { targeted_alliance = FRA enemy = GER } 
		}

			add_to_war = { targeted_alliance = FRA enemy = GER }
		}
		LUX = {
			add_to_war = { targeted_alliance = FRA enemy = GER }

	}
	
	option = {
		name = germany.1.c
		ai_chance = { factor = 100 }
		trigger = {
			country_exists = RUS
			NOT = {
				has_war_with = RUS
			}

		}
		create_wargoal = {
			target = RUS
			type = topple_government
		}
		declare_war_on = {
			target = RUS
			type = topple_government
		}
	}
	
	option = {
		name = germany.1.d
		ai_chance = { factor = 0 }
		FRA = {
			add_national_unity = 0.05
		}
		if = { 
			limit = {
				GER = { has_completed_focus = GER_respect_belgium }
			}
			RUS = {
				add_national_unity = 0.05
			}
		}	
	}
}
